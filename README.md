    public void view(){
        ArrayList<String> view = new ArrayList<String>();
        
        for (int i = 0; i<carrot; i++){
            view.add("Carrot");
        }
        for (int i = 0; i<tomato; i++){
            view.add("Tomato");
        }
        for (int i = 0; i<kebab; i++){
            view.add("Kebab");
        }
        String listString = "";
        for (String s : view){
            listString += s + "\n";
        } 
        
        jTextArea1.setText(listString);
    }
